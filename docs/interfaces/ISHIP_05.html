Title

ISHIP_05 - Decentralized File Storage

Notice

Main interface for IPFS file storage with encryption

interface ISHIP_05 {
    decryptData(
        encryptedData: string,
        options?: EncryptionOptions,
    ): Promise<string>;
    deleteFile(hash: string): Promise<{ error?: string; success: boolean }>;
    downloadFile(
        hash: string,
        options?: DownloadOptions,
    ): Promise<string | Blob>;
    encryptData(
        data: string | Buffer<ArrayBufferLike>,
        options?: EncryptionOptions,
    ): Promise<string>;
    getFileMetadata(hash: string): Promise<null | FileMetadata>;
    getIdentity(): ISHIP_00;
    getStorageStats(): Promise<
        {
            encryptedFiles: number;
            plainFiles: number;
            totalFiles: number;
            totalMB: number;
        },
    >;
    getUserFiles(): Promise<FileMetadata[]>;
    initialize(options?: SHIP_05_Config): Promise<void>;
    isFileAccessible(hash: string): Promise<boolean>;
    isInitialized(): boolean;
    uploadFile(
        file: Buffer<ArrayBufferLike> | File,
        options?: UploadOptions,
    ): Promise<UploadResult>;
    uploadJson(data: any, options?: UploadOptions): Promise<UploadResult>;
}
